   193→  const groups: Record<InfographicTemplateCategory, TemplateMeta[]> = {
   194→    list: [],
   195→    sequence: [],
   196→    compare: [],
   197→    hierarchy: [],
   198→    relation: [],
   199→    chart: [],

   201→  }
   202→  for (const t of TEMPLATE_REGISTRY) groups[t.category].push(t)
   203→  return groups
   204→}
